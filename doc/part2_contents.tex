    \item Some set of permissions.
    \item Contents

  \pause
  \vspace{5mm}

\defverbatim[colored]\contentRegexp{

import re

urls = [
    'https://www.gutenberg.org/files/11/11-h/11-h.htm',
    'https://golang.org',
    'http://www.google.com:80/',
    'definitely not a URL',
]
pattern = re.compile('([^:]+)://([^:/]+)(:\d+|)(/.*|)')

for url in urls:
    mo = pattern.match(url)
    
    if mo:
        print('proto="%s" domain="%s" port="%s" path="%s"' %
              (mo.group(1), mo.group(2), mo.group(3), mo.group(4)))
\end{minted}

\defverbatim[colored]\contentJSON{

>>> import json
>>> data = {'a': 1, 'b': [1,2,3], 'c': 'abc'}
>>> data
{'a': 1, 'b': [1, 2, 3], 'c': 'abc'}

>>> json.dumps(data)
'{"a": 1, "b": [1, 2, 3], "c": "abc"}'
>>> s = json.dumps(data, sort_keys=True, indent=4, separators=(',', ': '))
>>> s
'{\n    "a": 1,\n    "b": [\n        1,\n        2,\n        3\n    ],\n    "c": "abc"\n}'
>>> print(s)
{
    "a": 1,
    "b": [
        1,
        2,
        3
    ],
    "c": "abc"
}

>>> data2 = json.loads(s)
>>> data2
{'a': 1, 'b': [1, 2, 3], 'c': 'abc'}
\end{minted}

\subsection{Marshalling}
\begin{frame}[fragile]
    \frametitle{Exercises \subpart{Marshalling}}
    \vspace{3mm}
    Make some complex data structure that consists of integers, floats, booleans, strings, lists, tuples and dictionaries.
    
    \vspace{5mm}
    Marshall that to JSON and store the resulting string in a file.
    
    \vspace{5mm}
    Unmarshall the contents of that file to a new datastructure.
    
    \vspace{5mm}
    How similar is the new datastructure to the old?
\end{frame}

\subsection{CSV Corrections}
\begin{frame}[fragile]
    \frametitle{Exercises \subpart{CSV Corrections}}
    \vspace{-1mm}
    The CSV file format can hold tabular data. The format:
    \begin{itemize}
      \item Rows are separated by linebreaks.
      \item Columns are separated by commas.
    \end{itemize}
    
    \vspace{2mm}
    Excel is infamous for misinterpreting the CSV format. It uses semicolons instead of commas for separation.
    
    \vspace{2mm}
    The exercise:
    \begin{enumerate}
      \item Make a script that takes three parameters so that it can be called like this: \\
        \textcolor{purple}{\texttt{python3 csv\_reformatter.py fix input.csv output.csv}} \\
        \textcolor{purple}{\texttt{python3 csv\_reformatter.py unfix input.csv output.csv}}
      \item Decide on a reasonable intermediate format.
      \item Write functions for fundamental operations like reading, adjusting and writing data.
      \item Write the main code that parses the command line arguments, reads the input, adjusts it and writes the output.
      \item What could go wrong when converting between the two versions of the format?
    \end{enumerate}
\end{frame}

\subsection{Offloading to Other Processes}
\begin{frame}[fragile]
    \frametitle{Exercises \subpart{Offloading to Other Processes}}
    \vspace{3mm}
    Write a script called \texttt{read.py} that takes a filename as a parameter and prints out the contents of that file.
    
    \vspace{5mm}
    Write a script called \texttt{write.py} that takes a filename as a parameter and write whatever it gets on STDIN to this file.
    
    \vspace{5mm}
    Write a script called \texttt{filter.py} that takes a word as a parameter and forwards every line that gets from STDIN to STDOUT if the word is present.
    
    \vspace{5mm}
    Write a script that calls \texttt{read.py}, uses \texttt{filter.py} to filter the result and finally uses \texttt{write.py} to store the result.
    
    \vspace{5mm}
    Hint: \scalebox{0.9}{\url{https://docs.python.org/3/library/subprocess.html\#replacing-shell-pipeline}}
\end{frame}
